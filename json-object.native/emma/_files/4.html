548  /**
549   * Retrieve the native JSON properties information.
550   */
551  private Map<String, NativeJsonPropertyInfo> getNativeJsonPropertiesInfo() {
552    Map<String, NativeJsonPropertyInfo> nativeJsonPropertiesInfo = new HashMap<String, NativeJsonPropertyInfo>();
553    
554    Class<?> klass = this.getClass();
555    while (!klass.equals(NativeJsonObject.class)) {
556      for (Field field : klass.getDeclaredFields()) {
557        NativeJsonProperty nativeJsonPropertyAnnotation = field.getAnnotation(NativeJsonProperty.class);
558        if (nativeJsonPropertyAnnotation != null) {
559          String propertyName = field.getName();
560          if (nativeJsonPropertiesInfo.containsKey(propertyName)) {
561            throw new IllegalStateException("Duplicate native JSON property: " + propertyName);
562          }
563          
564          nativeJsonPropertiesInfo.put(propertyName, new NativeJsonPropertyInfo(field, nativeJsonPropertyAnnotation));
565        }
566      }
567      
568      klass = klass.getSuperclass();
569    }
570    
571    return nativeJsonPropertiesInfo;
572  }

576   * <p>

578   * native JSON property with the given name.
