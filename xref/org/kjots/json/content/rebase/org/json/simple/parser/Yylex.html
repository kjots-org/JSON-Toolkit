343   private boolean zzRefill() throws java.io.IOException {
344 
345     /* first: make room (if you can) */
346     if (zzStartRead > 0) {
347       System.arraycopy(zzBuffer, zzStartRead,
348                        zzBuffer, 0,
349                        zzEndRead-zzStartRead);
350 
351       /* translate stored positions */
352       zzEndRead-= zzStartRead;
353       zzCurrentPos-= zzStartRead;
354       zzMarkedPos-= zzStartRead;
355       zzStartRead = 0;
356     }
357 
358     /* is the buffer big enough? */
359     if (zzCurrentPos >= zzBuffer.length) {
360       /* if not: blow it up */
361       char newBuffer[] = new char[zzCurrentPos*2];
362       System.arraycopy(zzBuffer, 0, newBuffer, 0, zzBuffer.length);
363       zzBuffer = newBuffer;
364     }
365 
366     /* finally: fill the buffer with new input */
367     int numRead = zzReader.read(zzBuffer, zzEndRead,
368                                             zzBuffer.length-zzEndRead);
369 
370     if (numRead > 0) {
371       zzEndRead+= numRead;
372       return false;
373     }
374     // unlikely but not impossible: read 0 characters, but not at end of stream    
375     if (numRead == 0) {
376       int c = zzReader.read();
377       if (c == -1) {
378         return true;
379       } else {
380         zzBuffer[zzEndRead++] = (char) c;
381         return false;
382       }     
383     }
384 
385 	// numRead < 0
386     return true;
387   }
