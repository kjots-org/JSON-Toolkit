16  package org.kjots.json.object.ntive;
17  
18  import static org.junit.Assert.assertEquals;
19  
20  import org.junit.Before;
21  import org.junit.Test;
22  
23  /**
24   * Native JSON Number Map Test.
25   * <p>
26   * Created: 11th November 2010.
27   *
28   * @author <a href="mailto:kjots@kjots.org">Karl J. Ots &lt;kjots@kjots.org&gt;</a>
29   * @since 1.0
30   */
31  public class NativeJsonNumberMapTest {
32    /** The test native JSON number map. */
33    private NativeJsonNumberMap testNativeJsonNumberMap;
34    
35    /**
36     * Setup the test environment.
37     */
38    @Before
39    public void setup() {
40      this.testNativeJsonNumberMap = new NativeJsonNumberMap();
41    }
42    
43    /**
44     * Test the retrieval of an element of the map.
45     * <p>
46     * This test asserts that the native JSON number map correctly retrieves
47     * elements of the map.
48     */
49    @Test
50    public void testGet() {
51      for (int i = 0; i < 5; i++) {
52        testNativeJsonNumberMap.map.put(Integer.toString(i), Double.valueOf(i));
53      }
54      
55      for (int i = 0; i < 5; i++) {
56        assertEquals("testJsonNumberMap[\"" + Integer.toString(i) + "\"]", i, testNativeJsonNumberMap.get(Integer.toString(i)).doubleValue(), 0.1);
57      }
58    }
59    
60    /**
61     * Test the setting of an element of the map.
62     * <p>
63     * This test asserts that the native JSON number map correctly sets
64     * elements of the map.
65     */
66    @Test
67    public void testSet() {
68      for (int i = 0; i < 5; i++) {
69        testNativeJsonNumberMap.set(Integer.toString(i), Double.valueOf(i));
70      }
71      
72      for (int i = 0; i < 5; i++) {
73        assertEquals("object[\"" + Integer.toString(i) + "\"]", i, ((Number)testNativeJsonNumberMap.map.get(Integer.toString(i))).doubleValue(), 0.1);
74      }
75    }
76  }
